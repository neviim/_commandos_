        $ mosquitto_sub -h localhost -t /teste

        # terminal 2
        $ mosquitto_pub -h localhost -t /teste -m "teste 123"
        $ mosquitto_pub -h localhost -t /foo -m "teste 123"
        $ mosquitto_pub -h localhost -t /teste/bar -m "test123"
