# conda criar conta ambiente virtual

  ::: Ambiente windows: atualiza conda

    $ conda update -n base -c defaults conda

    $ conda update conda
    $ conda update pandas
    $ conda update numpy
    $ conda update matplotlib

    # lista ambientes virtuais instalados
    $ conda env list

  ::: cria virtual env

    $ conda create -n myenv python=3.4
    $ conda create -n myenv python
    $ conda create -n myenv scipy=0.15.0
    $ conda create -n myenv python=3.4 scipy=0.15.0 astroid babel

    $ conda env create -f environment.yml
    $ conda env list

    $ conda install -n myenv scipy=0.15.0
    $ conda install -n myenv scipy
    

    # outros exemplos
    $ cd d:\__CondaVirtual\
    $ conda create -n yourenvname python=x.x anaconda

      # install virtual area
      $ conda create -n pandas python=3.7 anaconda
      $ conda activate pandas
      $ conda deactivate


  ::: activando ambiente virtual

    $ conda activate py37
    $ conda deactivate

    $ source activate yourenvname
    $ conda install -n yourenvname [package]
    $ source deactivate

    $ conda activate myenv


  ::: remove

    $ conda remove -n yourenvname -all
    $ conda remove --name myenv --all


  ::: algumas bibliotecas relevantes

    $ conda install -c conda-forge geoplot
    $ conda install -c conda-forge geopandas

    $ pip install git+git://github.com/geopandas/geopandas.git

  ::: Explicit

    $ conda list --explicit
    $ conda list --explicit > spec-file.txt
